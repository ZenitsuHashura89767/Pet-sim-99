-- Логика фарма
_G.SoccerFarm = false

local function findField()
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and (v.Name:lower():find("soccer") or v.Name:lower():find("pitch")) then
            if v.Size.X > 50 and v.Size.Z > 50 then return v end
        end
    end
end

local function startFarm()
    task.spawn(function()
        local hum = game.Players.LocalPlayer.Character:FindFirstChild("Humanoid")
        local root = game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
        local field = findField()
        local startPos = field and field.Position or root.Position
        
        while _G.SoccerFarm do
            local z = (math.random(1, 20) * 5) - 40
            local target = startPos + Vector3.new(math.random(-80, 80), 0, z)
            hum:MoveTo(target)
            task.wait(1.5)
        end
    end)
end

ToggleButton.MouseButton1Click:Connect(function()
    _G.SoccerFarm = not _G.SoccerFarm
    ToggleButton.Text = _G.SoccerFarm and "WALK: ON" or "WALK: OFF"
    ToggleButton.BackgroundColor3 = _G.SoccerFarm and Color3.fromRGB(50, 180, 50) or Color3.fromRGB(200, 50, 50)
    if _G.SoccerFarm then startFarm() end
end)
